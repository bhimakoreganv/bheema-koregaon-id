// ========================================== // Bhima Koregaon Member Portal // register.js // Part 1 // ========================================== import { db } from "./firebase.js"; import { collection, getDocs, query, orderBy, limit, addDoc, serverTimestamp } from "https://www.gstatic.com/firebasejs/12.0.0/firebase-firestore.js"; // ========================= // Form Elements // ========================= const form = document.getElementById("registerForm"); const loading = document.getElementById("loading"); const successCard = document.getElementById("registerSuccess"); const successMessage = document.getElementById("successMessage"); const errorMessage = document.getElementById("errorMessage"); const generatedMemberId = document.getElementById("generatedMemberId"); // ========================= // Form Submit // ========================= form.addEventListener("submit", async (e) => { e.preventDefault(); loading.style.display = "block"; successMessage.style.display = "none"; errorMessage.style.display = "none"; try{ // ========================= // Read Form Values // ========================= const fullName = document.getElementById("fullName").value.trim(); const fatherName = document.getElementById("fatherName").value.trim(); const dob = document.getElementById("dob").value; const gender = document.getElementById("gender").value; const mobile = document.getElementById("mobile").value.trim(); const email = document.getElementById("email").value.trim(); const aadhaar = document.getElementById("aadhaar").value.trim(); const bloodGroup = document.getElementById("bloodGroup").value; const district = document.getElementById("district").value; const taluk = document.getElementById("taluk").value.trim(); const village = document.getElementById("village").value.trim(); const pincode = document.getElementById("pincode").value.trim(); const address = document.getElementById("address").value.trim(); const membershipType = document.getElementById("membershipType").value; const occupation = document.getElementById("occupation").value.trim(); const joiningDate = document.getElementById("joiningDate").value; const status = document.getElementById("status").value; const emergencyName = document.getElementById("emergencyName").value.trim(); const emergencyMobile = document.getElementById("emergencyMobile").value.trim(); const relation = document.getElementById("relation").value; const declaration = document.getElementById("declaration").checked; // ========================= // Validation // ========================= if( !fullName || !fatherName || !dob || !gender || !mobile || !district || !taluk || !village || !address || !membershipType || !declaration ){ throw new Error("Please fill all required fields."); } // ========================= // Generate Next Member ID // ========================= const membersRef = collection(db, "members"); const lastMemberQuery = query( membersRef, orderBy("memberId", "desc"), limit(1) ); const lastMemberSnapshot = await getDocs(lastMemberQuery); let memberId = "BK000001"; if (!lastMemberSnapshot.empty) { const lastMember = lastMemberSnapshot.docs[0].data(); const lastId = lastMember.memberId || "BK000000"; const lastNumber = parseInt(lastId.replace("BK", ""), 10); const nextNumber = lastNumber + 1; memberId = "BK" + String(nextNumber).padStart(6, "0"); } // ========================= // Member Data Object // ========================= const memberData = { memberId, fullName, fatherName, dob, gender, mobile, email, aadhaar, bloodGroup, district, taluk, village, pincode, address, membershipType, occupation, joiningDate, status, emergencyName, emergencyMobile, relation, createdAt: serverTimestamp() }; // ========================= // Save Member to Firestore // ========================= await addDoc( collection(db, "members"), memberData ); // ========================= // Hide Loading // ========================= loading.style.display = "none"; // ========================= // Show Success // ========================= generatedMemberId.textContent = memberId; successMessage.style.display = "block"; successCard.style.display = "block"; // ========================= // Reset Form // ========================= form.reset(); // Reset Photo Preview (Spark Version) const photoPreview = document.getElementById("photoPreview"); if(photoPreview){ photoPreview.src = "images/default-user.png"; } // Scroll to Success Card successCard.scrollIntoView({ behavior:"smooth" }); // ========================= // Error Handling // ========================= }catch(error){ console.error("Registration Error:", error); loading.style.display = "none"; errorMessage.style.display = "block"; errorMessage.innerHTML = "❌ " + error.message; errorMessage.scrollIntoView({ behavior:"smooth" }); } // ========================= // End Form Submit // ========================= }); // ========================================== // Helper Functions // ========================================== // Mobile Validation const mobileInput = document.getElementById("mobile"); mobileInput.addEventListener("input",()=>{ mobileInput.value = mobileInput.value.replace(/\D/g,""); }); // Aadhaar Validation const aadhaarInput = document.getElementById("aadhaar"); aadhaarInput.addEventListener("input",()=>{ aadhaarInput.value = aadhaarInput.value.replace(/\D/g,""); }); // Auto Joining Date const joiningDate = document.getElementById("joiningDate"); if(joiningDate.value===""){ const today = new Date(); joiningDate.value = today.toISOString().split("T")[0]; } // ========================================== // Utility Functions // ========================================== // Email Validation function isValidEmail(email){ const pattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/; return pattern.test(email); } // Mobile Validation function isValidMobile(mobile){ return /^[6-9]\d{9}$/.test(mobile); } // Aadhaar Validation function isValidAadhaar(aadhaar){ if(aadhaar==="") return true; return /^\d{12}$/.test(aadhaar); } // ========================= // Extra Validation // ========================= if(!isValidMobile(mobile)){ throw new Error( "Enter valid 10 digit mobile number." ); } if(email && !isValidEmail(email)){ throw new Error( "Invalid Email Address." ); } if(!isValidAadhaar(aadhaar)){ throw new Error( "Aadhaar must contain 12 digits." ); } // ========================= // Disable Register Button // ========================= const registerBtn = document.getElementById("registerBtn"); registerBtn.disabled = true; registerBtn.innerHTML = ' Registering...'; // ========================= // Enable Register Button // ========================= function enableRegisterButton(){ registerBtn.disabled = false; registerBtn.innerHTML = ' Register Member'; } // ========================= // Success Cleanup // ========================= setTimeout(()=>{ successMessage.style.display="none"; },5000); setTimeout(()=>{ errorMessage.style.display="none"; },5000); enableRegisterButton();
